    1. Create apps/client/src/app/profile/addresses/actions.ts (server actions):
       - Add 'use server' directive
       - Import { auth } from '@clerk/nextjs/server'
       - Import { prisma } from '@repo/db'
       - Import { revalidatePath } from 'next/cache'

       Export async function getAddresses():
       - Get { userId: clerkId } from auth(), throw if null
       - Find database user by clerkId
       - Return prisma.address.findMany({ where: { userId: dbUser.id }, orderBy: { isDefault: 'desc' } })

       Export async function createAddress(formData: FormData):
       - Get { userId: clerkId } from auth(), throw if null
       - Find database user by clerkId, throw 'User not found' if missing
       - Extract fields from formData: firstName, lastName, street, street2, city, state, zipCode, country, phone, label, isDefault
       - Validate required fields (firstName, lastName, street, city, state, zipCode, country)
       - If isDefault is true, run prisma.address.updateMany({ where: { userId: dbUser.id }, data: { isDefault: false } })
       - Create address: prisma.address.create({ data: { ...fields, userId: dbUser.id } })
       - revalidatePath('/profile/addresses')
       - Return { success: true }

       Export async function updateAddress(addressId: string, formData: FormData):
       - Get { userId: clerkId } from auth(), throw if null
       - Find database user by clerkId
       - Verify ownership: prisma.address.findFirst({ where: { id: addressId, userId: dbUser.id } })
       - Throw 'Address not found' if not owned by user
       - Extract and validate fields from formData
       - If isDefault is true, unset others first
       - prisma.address.update({ where: { id: addressId }, data: { ...fields } })
       - revalidatePath('/profile/addresses')
       - Return { success: true }

       Export async function deleteAddress(addressId: string):
       - Get { userId: clerkId } from auth(), throw if null
       - Find database user by clerkId
       - Verify ownership before delete
       - prisma.address.deleteMany({ where: { id: addressId, userId: dbUser.id } })
       - revalidatePath('/profile/addresses')
       - Return { success: true }

       Export async function setDefaultAddress(addressId: string):
       - Get auth, find user
       - Verify ownership
       - Transaction: unset all defaults, set this one as default
       - revalidatePath('/profile/addresses')
