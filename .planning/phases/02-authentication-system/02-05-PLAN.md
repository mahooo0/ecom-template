    1. Create apps/admin/src/app/dashboard/users/actions.ts:
       - Add 'use server' directive

       - Import { prisma } from '@repo/db'
       - Import { revalidatePath } from 'next/cache'
       - Import { redirect } from 'next/navigation'

       Helper function (not exported) to verify admin access:
       ```typescript
       async function verifyAdmin() {
         const { sessionClaims } = await auth();
         const role = sessionClaims?.metadata?.role;
         if (role !== 'ADMIN' && role !== 'SUPER_ADMIN') {
           redirect('/unauthorized');
         }
         return role;
       }
       ```

       Export async function getUsers(page = 1, limit = 20):
       - Call verifyAdmin()
       - Query prisma.user.findMany with skip/take pagination, orderBy createdAt desc
       - Include _count for addresses, reviews, wishlists
       - Also get total count for pagination
       - Return { users, total, page, limit, totalPages }

       Export async function getUserDetail(userId: string):
       - Call verifyAdmin()
       - Query prisma.user.findUnique({ where: { id: userId }, include: { addresses: true, _count: { select: { reviews: true, wishlists: true } } } })
       - Also get the Clerk user for status info: clerkClient().users.getUser(dbUser.clerkId)
       - Return combined data: database user with addresses + Clerk user banned status

       Export async function setUserRole(userId: string, newRole: 'CUSTOMER' | 'ADMIN' | 'SUPER_ADMIN'):
       - Call verifyAdmin()
       - Get database user by id to get clerkId
       - Call clerkClient().users.updateUserMetadata(dbUser.clerkId, { publicMetadata: { role: newRole } })
       - Also update local database: prisma.user.update({ where: { id: userId }, data: { role: newRole } })
         (This provides immediate consistency; webhook will also fire but upsert is idempotent)
       - revalidatePath('/dashboard/users')
       - revalidatePath(`/dashboard/users/${userId}`)
       - Return { success: true }

       Export async function toggleUserStatus(userId: string, shouldBan: boolean):
       - Call verifyAdmin()
       - Get database user by id to get clerkId
       - If shouldBan: call clerkClient().users.banUser(dbUser.clerkId)
       - If !shouldBan: call clerkClient().users.unbanUser(dbUser.clerkId)
       - Update local database: prisma.user.update({ where: { id: userId }, data: { isActive: !shouldBan } })
       - revalidatePath('/dashboard/users')
       - revalidatePath(`/dashboard/users/${userId}`)
       - Return { success: true }
  </action>
